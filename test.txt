 let showPerPage = 1;
let currentPage = 0;
let activate = 2;
let numberOfPages;
let minRange = 0;

// Set css diplay block per page
function setDisplay(first, last) {
    $('body #pagination').children().css('display', 'none');
    $('body #pagination').children().slice(first, last).css('display', 'block');
}

// Previous function
function previous(){
    goToPage(currentPage - 1);
}

// Next function
function next(){
    goToPage(currentPage + 1);
}

function doNum(currentPage){
    if (currentPage < 3) {
        minRange = 0;
        let navLink = 0;
        for (let i = 2; i < 7; i++) {
            $( ".pagination li:nth-child("+i+") a" ).html(navLink+1);
            $( ".pagination li:nth-child("+i+") a" ).attr("href", "javascript:goToPage("+navLink+")");
            navLink++;
        }
    }
    
    let navButton = currentPage-2;
    if (currentPage > 2) {
        for (let i = 2; i < 7; i++) {
            $( ".pagination li:nth-child("+i+") a" ).html(navButton+1);
            $( ".pagination li:nth-child("+i+") a" ).attr("href", "javascript:goToPage("+navButton+")");
            navButton++;
            minRange = navButton-5;
        }
    }
}

//Go to page number
function goToPage(pageNumber){
    currentPage = pageNumber;
    startFrom = currentPage * showPerPage;
    endOn = startFrom + showPerPage;
    setDisplay(startFrom, endOn);
    $( ".pagination .active").removeClass('active');
    if (currentPage < 3) {
        activate = currentPage+2;
    } else {
        activate = 4;
    }
    doNum(currentPage);
    $( ".pagination :nth-child("+activate+")").addClass('active');
    update(currentPage);
} 

// Update pagination
function update(currentPage){

// On / Off Prev. Next.
if (currentPage === 0) {
    $( ".pagination li:nth-child(1)" ).addClass('disabled').attr("tabindex", "-1");
} else {
    $( ".pagination li:nth-child(1)" ).removeClass('disabled').removeAttr("tabindex");
}
let disableOffLimit = numberOfPages;
if (currentPage >= 2) {
    disableOffLimit = disableOffLimit-minRange;
} else {
    disableOffLimit = numberOfPages;
}
if (currentPage != numberOfPages) {
        $( ".pagination li:gt("+disableOffLimit+"):not(li:last)" ).addClass('disabled').attr("tabindex", "-1");
    }
if (currentPage+1 === numberOfPages){
        $( ".pagination li:last" ).addClass('disabled').attr("tabindex", "-1");
    } 
}
